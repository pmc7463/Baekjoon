#include <stdio.h>

int main(void) {
    int material, hate[100], pizza;
    int tempMaterial[100], tempHate[100][100];
    int i, j;
    int flag = 0, counter = 0;
    scanf("%d", &material);

    for (i = 0; i < material; i++) 
        scanf("%d", &hate[i]);
    
    scanf("%d", &pizza);

    for(i = 0; i < pizza; i++) {
        scanf("%d", &tempMaterial[i]);
            for (j = 0; j < tempMaterial[i]; j++)
                scanf("%d", &tempHate[i][j]);
    }
    
    for (i = 0; i < pizza; i++) {
        flag = 0;
        for (j = 0; j < tempMaterial[i]; j++) {
            if (hate[j] == tempHate[i][j])
                if(flag < 1)
                    flag++;
        }
        if (flag == 1)
            counter++;
    }
    printf("%d\n", pizza - counter);

    return 0;
}

#include <stdio.h>


int main(void)
{
    int arr[7];
    int sum = 0, min[7], temp;
    for (int i = 0; i < 7; i++) {
        scanf("%d", &arr[i]);
        if (arr[i]%2 != 0) {
            sum += arr[i];
            min[7] = arr[i];
        }
    }

    temp = min[0];

    for (int i = 1; i < sizeof(min); i++) {
        if (temp > min[i])
            temp = min[i];
    }

    printf("%d\n", sum);
    printf("%d\n", temp);

    return 0;
}
